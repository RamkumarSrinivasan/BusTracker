1 /**
  2  * this represents the mobile device, and provides properties for inspecting the model, version, UUID of the
  3  * phone, etc.
  4  * @constructor
  5  */
  6 function Device() {
  7     this.available = PhoneGap.available;
  8     this.model     = null;
  9     this.version   = null;
 10     this.gap       = null;
 11     this.uuid      = null;
 12     try {
 13         if (window['DroidGap'] != undefined && window.DroidGap.exists()) {
 14             this.available = true;
 15             this.isAndroid = true;
 16             this.uuid = window.DroidGap.getUuid();
 17             this.gapVersion = window.DroidGap.getVersion();
 18         } else {          
 19             this.model     = DeviceInfo.platform;
 20             this.version   = DeviceInfo.version;
 21             this.gap       = DeviceInfo.gap;
 22             this.uuid      = DeviceInfo.uuid;
 23         }
 24     } catch(e) {
 25         this.available = false;
 26     }
 27 }
 28 
 29 PhoneGap.addConstructor(function() {
 30     navigator.device = window.device = new Device();
 31 });
