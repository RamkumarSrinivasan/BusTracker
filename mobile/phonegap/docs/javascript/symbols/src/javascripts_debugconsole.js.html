  8 /**
  9  * Utility function for rendering and indenting strings, or serializing
 10  * objects to a string capable of being printed to the console.
 11  * @param {Object|String} message The string or object to convert to an indented string
 12  * @private
 13  */
 14 DebugConsole.prototype.processMessage = function(message) {
 15     if (typeof(message) != 'object') {
 16         return message;
 17     } else {
 18         /**
 19          * @function
 20          * @ignore
 21          */
 22         function indent(str) {
 23             return str.replace(/^/mg, "    ");
 24         }
 25         /**
 26          * @function
 27          * @ignore
 28          */
 29         function makeStructured(obj) {
 30             var str = "";
 31             for (var i in obj) {
 32                 try {
 33                     if (typeof(obj[i]) == 'object') {
 34                         str += i + ":\n" + indent(makeStructured(obj[i])) + "\n";
 35                     } else {
 36                         str += i + " = " + indent(String(obj[i])).replace(/^    /, "") + "\n";
 37                     }
 38                 } catch(e) {
 39                     str += i + " = EXCEPTION: " + e.message + "\n";
 40                 }
 41             }
 42             return str;
 43         }
 44         return "Object:\n" + makeStructured(message);
 45     }
 46 };
