1 /**
  2  * This class provides access to device GPS data.
  3  * @constructor
  4  */
  5 function Geolocation() {
  6     /**
  7      * The last known GPS position.
  8      */
  9     this.lastPosition = null;
 10     this.lastError = null;
 11     this.callbacks = {
 12         onLocationChanged: [],
 13         onError:           []
 14     };
 15 };
 16 
 17 /**
 18  * Asynchronously aquires the current position.
 19  * @param {Function} successCallback The function to call when the position
 20  * data is available
 21  * @param {Function} errorCallback The function to call when there is an error 
 22  * getting the position data.
 23  * @param {PositionOptions} options The options for getting the position data
 24  * such as timeout.
 25  */
 26 Geolocation.prototype.getCurrentPosition = function(successCallback, errorCallback, options) {
 27     var referenceTime = 0;
 28     if (this.lastPosition)
 29         referenceTime = this.lastPosition.timeout;
 30     else
 31         this.start(options);
 32 
 33     var timeout = 20000;
 34     var interval = 500;
 35     if (typeof(options) == 'object' && options.interval)
 36         interval = options.interval;
 37 
 38     if (typeof(successCallback) != 'function')
 39         successCallback = function() {};
 40     if (typeof(errorCallback) != 'function')
 41         errorCallback = function() {};
 42 
 43     var dis = this;
 44     var delay = 0;
 45     var timer = setInterval(function() {
 46         delay += interval;
 47 
 48         if (typeof(dis.lastPosition) == 'object' && dis.lastPosition.timestamp > referenceTime) {
 49             successCallback(dis.lastPosition);
 50             clearInterval(timer);
 51         } else if (delay >= timeout) {
 52             errorCallback();
 53             clearInterval(timer);
 54         }
 55     }, interval);
 56 };
 57 
 58 /**
 59  * Asynchronously aquires the position repeatedly at a given interval.
 60  * @param {Function} successCallback The function to call each time the position
 61  * data is available
 62  * @param {Function} errorCallback The function to call when there is an error 
 63  * getting the position data.
 64  * @param {PositionOptions} options The options for getting the position data
 65  * such as timeout and the frequency of the watch.
 66  */
 67 Geolocation.prototype.watchPosition = function(successCallback, errorCallback, options) {
 68 	// Invoke the appropriate callback with a new Position object every time the implementation 
 69 	// determines that the position of the hosting device has changed. 
 70 	
 71 	this.getCurrentPosition(successCallback, errorCallback, options);
 72 	var frequency = 10000;
 73         if (typeof(options) == 'object' && options.frequency)
 74             frequency = options.frequency;
 75 	
 76 	var that = this;
 77 	return setInterval(function() {
 78 		that.getCurrentPosition(successCallback, errorCallback, options);
 79 	}, frequency);
 80 };

 83 /**
 84  * Clears the specified position watch.
 85  * @param {String} watchId The ID of the watch returned from #watchPosition.
 86  */
 87 Geolocation.prototype.clearWatch = function(watchId) {
 88 	clearInterval(watchId);
 89 };
 90 
 91 /**
 92  * Called by the geolocation framework when the current location is found.
 93  * @param {PositionOptions} position The current position.
 94  */
 95 Geolocation.prototype.setLocation = function(position) {
 96     this.lastPosition = position;
 97     for (var i = 0; i < this.callbacks.onLocationChanged.length; i++) {
 98         var f = this.callbacks.onLocationChanged.shift();
 99         f(position);
100     }
101 };
102 
103 /**
104  * Called by the geolocation framework when an error occurs while looking up the current position.
105  * @param {String} message The text of the error message.
106  */
107 Geolocation.prototype.setError = function(message) {
108     this.lastError = message;
109     for (var i = 0; i < this.callbacks.onError.length; i++) {
110         var f = this.callbacks.onError.shift();
111         f(message);
112     }
113 };
