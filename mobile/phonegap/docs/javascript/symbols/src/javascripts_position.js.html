1 /**
  2  * This class contains position information.
  3  * @param {Object} lat
  4  * @param {Object} lng
  5  * @param {Object} acc
  6  * @param {Object} alt
  7  * @param {Object} altacc
  8  * @param {Object} head
  9  * @param {Object} vel
 10  * @constructor
 11  */
 12 function Position(lat, lng, acc, alt, altacc, head, vel) {
 13 	/**
 14 	 * The latitude of the position.
 15 	 */
 16 	this.latitude = lat;
 17 	/**
 18 	 * The longitude of the position,
 19 	 */
 20 	this.longitude = lng;
 21 	/**
 22 	 * The accuracy of the position.
 23 	 */
 24 	this.accuracy = acc;
 25 	/**
 26 	 * The altitude of the position.
 27 	 */
 28 	this.altitude = alt;
 29 	/**
 30 	 * The altitude accuracy of the position.
 31 	 */
 32 	this.altitudeAccuracy = altacc;
 33 	/**
 34 	 * The direction the device is moving at the position.
 35 	 */
 36 	this.heading = head;
 37 	/**
 38 	 * The velocity with which the device is moving at the position.
 39 	 */
 40 	this.velocity = vel;
 41 	/**
 42 	 * The time that the position was obtained.
 43 	 */
 44 	this.timestamp = new Date().getTime();
 45 }
 46 
 47 /**
 48  * This class specifies the options for requesting position data.
 49  * @constructor
 50  */
 51 function PositionOptions() {
 52 	/**
 53 	 * Specifies the desired position accuracy.
 54 	 */
 55 	this.enableHighAccuracy = true;
 56 	/**
 57 	 * The timeout after which if position data cannot be obtained the errorCallback
 58 	 * is called.
 59 	 */
 60 	this.timeout = 10000;
 61 }
 62 
 63 /**
 64  * This class contains information about any GSP errors.
 65  * @constructor
 66  */
 67 function PositionError() {
 68 	this.code = null;
 69 	this.message = "";
 70 }
 71 
 72 PositionError.UNKNOWN_ERROR = 0;
 73 PositionError.PERMISSION_DENIED = 1;
 74 PositionError.POSITION_UNAVAILABLE = 2;
 75 PositionError.TIMEOUT = 3;
