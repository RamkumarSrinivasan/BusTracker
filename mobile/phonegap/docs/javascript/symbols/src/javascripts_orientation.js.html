 24 /**
 25  * Asynchronously aquires the orientation repeatedly at a given interval.
 26  * @param {Function} successCallback The function to call each time the orientation
 27  * data is available.
 28  * @param {Function} errorCallback The function to call when there is an error 
 29  * getting the orientation data.
 30  */
 31 Orientation.prototype.watchOrientation = function(successCallback, errorCallback) {
 32 	// Invoke the appropriate callback with a new Position object every time the implementation 
 33 	// determines that the position of the hosting device has changed. 
 34 	this.getCurrentPosition(successCallback, errorCallback);
 35 	return setInterval(function() {
 36 		navigator.orientation.getCurrentOrientation(successCallback, errorCallback);
 37 	}, 10000);
 38 }

 40 /**
 41  * Clears the specified orientation watch.
 42  * @param {String} watchId The ID of the watch returned from #watchOrientation.
 43  */
 44 Orientation.prototype.clearWatch = function(watchId) {
 45 	clearInterval(watchId);
 46 }
