 82 /**
 83  * Function called when a tab bar item has been selected.
 84  * @param {Number} tag the tag number for the item that has been selected
 85  */
 86 UIControls.prototype.tabBarItemSelected = function(tag) {
 87     if (typeof(this.tabBarCallbacks[tag]) == 'function')
 88         this.tabBarCallbacks[tag]();
 89 };
