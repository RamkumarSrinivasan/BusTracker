1 /**
  2  * This class provides access to device accelerometer data.
  3  * @constructor
  4  */
  5 function Accelerometer() {
  6 	/**
  7 	 * The last known acceleration.
  8 	 */
  9 	this.lastAcceleration = null;
 10 }
 11 
 12 /**
 13  * Asynchronously aquires the current acceleration.
 14  * @param {Function} successCallback The function to call when the acceleration
 15  * data is available
 16  * @param {Function} errorCallback The function to call when there is an error 
 17  * getting the acceleration data.
 18  * @param {AccelerationOptions} options The options for getting the accelerometer data
 19  * such as timeout.
 20  */
 21 Accelerometer.prototype.getCurrentAcceleration = function(successCallback, errorCallback, options) {
 22 	// If the acceleration is available then call success
 23 	// If the acceleration is not available then call error
 24 	
 25 	// Created for iPhone, Iphone passes back _accel obj litteral
 26 	if (typeof successCallback == "function") {
 27 		var accel = new Acceleration(_accel.x,_accel.y,_accel.z);
 28 		Accelerometer.lastAcceleration = accel;
 29 		successCallback(accel);
 30 	}
 31 }
 32 
 33 /**
 34  * Asynchronously aquires the acceleration repeatedly at a given interval.
 35  * @param {Function} successCallback The function to call each time the acceleration
 36  * data is available
 37  * @param {Function} errorCallback The function to call when there is an error 
 38  * getting the acceleration data.
 39  * @param {AccelerationOptions} options The options for getting the accelerometer data
 40  * such as timeout.
 41  */
 42 
 43 Accelerometer.prototype.watchAcceleration = function(successCallback, errorCallback, options) {
 44 	this.getCurrentAcceleration(successCallback, errorCallback, options);
 45 	// TODO: add the interval id to a list so we can clear all watches
 46  	var frequency = (options != undefined)? options.frequency : 10000;
 47 	return setInterval(function() {
 48 		navigator.accelerometer.getCurrentAcceleration(successCallback, errorCallback, options);
 49 	}, frequency);
 50 }
 51 
 52 /**
 53  * Clears the specified accelerometer watch.
 54  * @param {String} watchId The ID of the watch returned from #watchAcceleration.
 55  */
 56 Accelerometer.prototype.clearWatch = function(watchId) {
 57 	clearInterval(watchId);
 58 }

 61     if (typeof navigator.accelerometer == "undefined") navigator.accelerometer = new Accelerometer();
